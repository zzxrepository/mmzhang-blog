class Solution {
	public int[] findMode(TreeNode root) {
        //å¦‚æœä¸è€ƒè™‘äºŒå‰æœç´¢æ ‘ï¼Œä¸€èˆ¬çš„è§£æ³•
        //åˆ›å»ºmapé›†åˆç”¨äºè®°å½•æ‰€æœ‰å…ƒç´ å‡ºç°çš„é¢‘æ¬¡

		List<Integer> list = new ArrayList<>();
        //åˆ¤æ–­ç‰¹æ®Šæƒ…å†µ

            return list.stream().mapToInt(Integer::intValue).toArray();

		// éå†äºŒå‰æ ‘ï¼Œè·å¾—æ‰€æœ‰å…ƒç´ å‡ºç°çš„æ¬¡æ•°
		searchBST(root, map);

		List<Map.Entry<Integer, Integer>> mapList = map.entrySet().stream()

				.collect(Collectors.toList());
        //è·å–å‡ºç°æœ€é«˜é¢‘æ¬¡çš„å…ƒç´ 
		list.add(mapList.get(0).getKey());

		for(int i = 1; i < mapList.size(); i++){

				list.add(mapList.get(i).getKey());
			} else {
				break;
			}
		}
        //å°†åˆ—è¡¨è½¬åŒ–æˆæ•°ç»„è¿”å›
		return list.stream().mapToInt(Integer::intValue).toArray();
	}

    //é€’å½’æ³• å‰åºéå†
	public void searchBST(TreeNode cur, Map<Integer, Integer> map) {
		if (cur == null) return;
        //å¦‚æœæ²¡å‡ºç°è¿‡å°±æŒ‡å®šä¸º0ï¼šmap.getOrDefault(cur.val, 0)
		map.put(cur.val, map.getOrDefault(cur.val, 0) + 1);
		searchBST(cur.left, map);
		searchBST(cur.right, map);
	}
}
